## Column Header Sorting

Clicking a column header now toggles through sort states:

```tsx title="Sort via column header"
function ColumnHeader({ column }: { column: Column }) {
  const [sortState, setSortState] = useState<'none' | 'asc' | 'desc'>('none')
  
  const handleClick = () => {
    const nextState = {
      'none': 'asc',
      'asc': 'desc',
      'desc': 'none',
    }[sortState]
    setSortState(nextState)
    column.toggleSorting(nextState === 'desc')
  }
  
  return (
    <button onClick={handleClick} className="flex items-center gap-1">
      {column.name}
      {sortState === 'asc' && <ArrowUpIcon className="h-4 w-4" />}
      {sortState === 'desc' && <ArrowDownIcon className="h-4 w-4" />}
    </button>
  )
}
```

## Column Visibility and Order

Users can now customize which columns are visible and their order:

```tsx title="Column visibility popover"
<Popover>
  <PopoverTrigger asChild>
    <Button variant="outline" size="sm">
      <ColumnsIcon className="mr-2 h-4 w-4" />
      Columns
    </Button>
  </PopoverTrigger>
  <PopoverContent className="w-56">
    <DragDropContext onDragEnd={handleDragEnd}>
      <Droppable droppableId="columns">
        {(provided) => (
          <div {...provided.droppableProps} ref={provided.innerRef}>
            {columns.map((column, index) => (
              <Draggable key={column.id} draggableId={column.id} index={index}>
                {(provided) => (
                  <div
                    ref={provided.innerRef}
                    {...provided.draggableProps}
                    className="flex items-center gap-2 py-1"
                  >
                    <span {...provided.dragHandleProps}>
                      <GripVerticalIcon className="h-4 w-4" />
                    </span>
                    <Checkbox
                      checked={column.visible}
                      onCheckedChange={(checked) => 
                        toggleColumnVisibility(column.id, checked)
                      }
                    />
                    <span>{column.name}</span>
                  </div>
                )}
              </Draggable>
            ))}
            {provided.placeholder}
          </div>
        )}
      </Droppable>
    </DragDropContext>
  </PopoverContent>
</Popover>
```

## Filter and Sort Logic Refactoring

I moved the filter and sort logic into dedicated hooks for better organization:

```typescript title="useTableFilters hook"
function useTableFilters(initialFilters: Filter[]) {
  const [filters, setFilters] = useState(initialFilters)
  
  const addFilter = useCallback((filter: Filter) => {
    setFilters((prev) => [...prev, filter])
  }, [])
  
  const removeFilter = useCallback((id: string) => {
    setFilters((prev) => prev.filter((f) => f.id !== id))
  }, [])
  
  const updateFilter = useCallback((id: string, updates: Partial<Filter>) => {
    setFilters((prev) =>
      prev.map((f) => (f.id === id ? { ...f, ...updates } : f))
    )
  }, [])
  
  const buildFilterQuery = useCallback(() => {
    return filters
      .filter((f) => f.value !== undefined)
      .map((f) => `${f.column}.${f.operator}.${f.value}`)
      .join(',')
  }, [filters])
  
  return { filters, addFilter, removeFilter, updateFilter, buildFilterQuery }
}
```

## Optimistic Updates

Filter and sort updates now apply optimistically:

```typescript title="Optimistic filter updates"
const applyFilter = async (filter: Filter) => {
  // Optimistically update UI
  setFilters((prev) => [...prev, filter])
  
  try {
    // Sync with server
    await updateTableState({ filters: [...filters, filter] })
  } catch (error) {
    // Rollback on error
    setFilters((prev) => prev.filter((f) => f.id !== filter.id))
    toast.error('Failed to apply filter')
  }
}
```
